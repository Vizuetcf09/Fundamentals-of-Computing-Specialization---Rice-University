import math

# intialize globals
width = 450
height = 300
ball_list = []
ball_radius = 15
ball_color = "Red"

# -----

# define event handler for mouse click, draw
def click(pos):
    remove = []
    
    for ball in ball_list:
        if distance(ball, pos) < ball_radius:
            remove.append(ball)

        if remove == []:
            ball_list.append(pos)
    
        else:
            for ball in remove:
                ball_list.pop(ball_list.index(ball))
