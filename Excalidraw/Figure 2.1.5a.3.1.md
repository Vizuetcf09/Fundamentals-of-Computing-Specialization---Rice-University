# Simple task list

import simplegui

tasks = []

# Handler for button
def clear():
    global tasks
    tasks = []
    
# Handler for new task
def new(task):
    tasks.append(task)

# -----

# Create a frame and assign callbacks 
# to event handlers
frame = simplegui.create_frame("Task List", 
                                        600, 
                                        400)
frame.add_input("New task:", new, 200)
