# define event handler for mouse click, draw
def click(pos):
    changed = False
    
    for ball in ball_list
        if distance([ball[0], ball[1]], pos) < ball_radius:
                ball[2] = "Green"
                changed = True
    
    if not changed:
        ball_list.append([pos[0], pos[1], "Red"])
